# Create API ongkir on your own website

// Include library ongkir
require("lib/ongkir.php");

// Create objek ongkir(dari, ke, berat)
$ongkir = new ongkir('bandung', 'surabaya', 5);

// Menampilkan ongkir JNE
echo $ongkir-&gt;jne();

/* Menampilkan ongkir TIKI
echo $ongkir-&gt;tiki();
*/
